    ```ts
    "use server";

    import { signIn } from "@/lib/auth";
    import { loginSchema } from "@/lib/validators/auth";
    import { AuthError } from "next-auth";

    export type LoginActionResult =
      | { ok: true }
      | { ok: false; error: string };

    export async function loginWithCredentials(
      formData: FormData
    ): Promise<LoginActionResult> {
      const parsed = loginSchema.safeParse({
        email: formData.get("email"),
        password: formData.get("password"),
      });
      if (!parsed.success) {
        return { ok: false, error: "Dados inválidos" };
      }
      try {
        await signIn("credentials", {
          ...parsed.data,
          redirect: false,
        });
        return { ok: true };
      } catch (e) {
        if (e instanceof AuthError) {
          return { ok: false, error: "Credenciais inválidas" };
        }
        throw e;
      }
    }
    ```

    ```ts
    "use server";

    import bcrypt from "bcryptjs";
    import { db } from "@/lib/db";
    import { signupSchema } from "@/lib/validators/auth";
    import { signIn } from "@/lib/auth";

    export type SignupActionResult =
      | { ok: true }
      | { ok: false; error: string };

    export async function createAccount(
      formData: FormData
    ): Promise<SignupActionResult> {
      const parsed = signupSchema.safeParse({
        name: formData.get("name"),
        email: formData.get("email"),
        password: formData.get("password"),
        confirmPassword: formData.get("confirmPassword"),
      });
      if (!parsed.success) {
        return {
          ok: false,
          error: parsed.error.issues[0]?.message ?? "Dados inválidos",
        };
      }

      const { name, email, password } = parsed.data;

      // Evitar colisão de email — tratamos UNIQUE violation explícito
      const existing = await db.user.findUnique({ where: { email } });
      if (existing) {
        return { ok: false, error: "Já existe uma conta com esse e-mail" };
      }

      const hash = await bcrypt.hash(password, 10);

      await db.user.create({
        data: {
          name,
          email,
          password: hash,
          role: "CUSTOMER",
        },
      });

      // Auto-login pós-cadastro
      try {
        await signIn("credentials", { email, password, redirect: false });
      } catch {
        // Se o auto-login falhar, o cadastro já foi criado — o client redireciona para /login
      }

      return { ok: true };
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client (navegador) → Server Action | Toda action DEVE começar com auth() — não confiar que "está no servidor" |
| anonymous → /admin/* | 3 barreiras: middleware (edge), layout RSC, requireAdmin() em cada action |
| anonymous → /pedidos, /perfil, /cliente/* | 2 barreiras: middleware + requireAuth() |
| request → Credentials provider | bcrypt.compare é timing-safe; senhas nunca entram em logs (Sentry PII scrub do Plan 04) |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-00-08 | E (Elevation of Privilege) | Server Actions expostas como POST público | mitigate | requireAdmin() em toda action admin + tests (Pitfall #8 — cobertura de regressão no CI) |
| T-00-09 | I (Information Disclosure) | Existência do painel admin | mitigate | `notFound()` em vez de `forbidden` — não revela a rota para scanner |
| T-00-10 | S (Spoofing) | Tentativa de login com senha errada | mitigate | bcrypt.compare no authorize de lib/auth.ts + mensagem genérica "Credenciais inválidas" (não revela se o email existe) |
| T-00-11 | T (Tampering) | Token JWT — role escalation via client | mitigate | `jwt({token, user})` só lê role do user na primeira autorização; não aceita role vinda do client |
| T-00-12 | I (Information Disclosure) | Senha em plaintext em log/erro | mitigate | password NUNCA retorna de findUnique sem select explícito; bcrypt hash, 10 rounds, nunca serializado |
</threat_model>
